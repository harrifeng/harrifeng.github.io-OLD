    def get_min_count(S):
    minc = [sys.maxint] * (S + 1)
    minc[0] = 0
    val = [1, 2, 5]
    for i in xrange(S+1):
        for v in val:
            if i >= v and minc[i-v] + 1 < minc[i]:
                minc[i] = minc[i-v] + 1
    return minc[S]
