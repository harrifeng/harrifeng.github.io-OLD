#include <iostream>
using namespace std;

int binary_search_first(int arr[], int len, int key) {
    int low = 0;
    int high = len - 1;
    int result = -1;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (key == arr[mid]) {
            result = mid;
            high = mid -1;
        } else if (key < arr[mid]) {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return result;
}

int binary_search_last(int arr[], int len, int key) {
    int low = 0;
    int high = len - 1;
    int result = -1;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (arr[mid] == key) {
            result = mid;
            low = mid + 1;
        } else if (arr[mid] > key) {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return result;
}

int main(int argc, char *argv[])
{
    int array[] = {0,0,1,2,4,5,5,6,6,6,6,7,11};
    int n = sizeof(array) / sizeof(array[0]);
    cout << "Found 5 from " << binary_search_first(array, n, 5) << " to "
         << binary_search_last(array, n, 5) << endl;

    cout << "Found 8 from " << binary_search_first(array, n, 8) << " to "
         << binary_search_last(array, n, 8) << endl;

    return 0;
}
