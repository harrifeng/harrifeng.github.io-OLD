def max_heapify(A, root, end):
    left = root * 2 + 1
    right = root * 2 + 2
    max = root
    if left <= end and A[left] > A[root]:
        max = left
    if right <= end and A[right] > A[max]:
        max = right
    if max != root:
        A[max], A[root] = A[root], A[max]
        max_heapify(A, max, end)


def build_max_heap(A):
    size = len(A) - 1
    for i in reversed(range(0, size/2)):
        max_heapify(A, i, size)


def heap_sort(A):

import random
import heapsort


def top_n(lst, n):
    """ list contains very large number of input, n
    is relatively small number, for intance 100.
    Solve the question like "find top 10 min elements from billion"
    """
    nlist = lst[:n]
    build_max_heap(nlist)

    for num in lst[n:]:
        if num < nlist[0]:
            nlist[0] = num
            build_max_heap(nlist)
    return nlist


if __name__ == '__main__':
    aa = range(10**9)
    random.shuffle(aa)
