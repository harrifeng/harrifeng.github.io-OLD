struct Number {
    int index;
    int value;
    Number(int i, int v) : index(i),  value(v){}
};

bool compare_number(const Number &left, const Number &right) {
    return left.value < right.value;
}

class Solution {
public:
    vector<int> twoSum(vector<int> &numbers, int target) {
        vector<Number> nn;
        for (int i = 0; i < numbers.size(); i++) {
            nn.push_back(Number(i+1, numbers[i]));
        }
        std::sort(nn.begin(), nn.end(), compare_number);
        int beg = 0;
        int end = nn.size() - 1;

        while (beg < end) {
            int sum = nn[beg].value + nn[end].value;
            if (sum == target) {
                int small = std::min(nn[beg].index, nn[end].index);
                int big = std::max(nn[beg].index, nn[end].index);
                vector<int> ret;
                ret.push_back(small);
                ret.push_back(big);
                return ret;
            } else if (sum < target) {
                beg++;
            } else {
                end--;
            }
        }
    }
};
