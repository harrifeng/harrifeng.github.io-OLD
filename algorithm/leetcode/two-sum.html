Leetcode 001: Two Sum

Table of Contents

  • 题目
  • 思路
  • 解答

题目

  • 给出一组括号字符串,比如{2, 7, 11, 15}和一个目标数9, 判断这些数中哪些数字的 和为目标数, 输出他们的下标 
Sample Input : {2, 7, 11, 15}, target=9
Sample Output: 1, 2

思路

  • 求哪些数字和为某个特定的target很简单,就是两个指针从两头指的问题, 但是要求 下标的话,就麻烦一些. 因为两指针法需要排序,而排序后很难找到原始下标
  • 这里处理的方法很OOP, 每个数字都用原始下标和数值组合成一个新的类
  • 注意这里compare_number和std::sort的应用
  • 还可以使用Java里面的HashMap,那解决起来就太简单.有点无耻.

解答

struct Number {
    int index;
    int value;
    Number(int i, int v) : index(i),  value(v){}
};

bool compare_number(const Number &left, const Number &right) {
    return left.value < right.value;
}

class Solution {
public:
    vector<int> twoSum(vector<int> &numbers, int target) {
        vector<Number> nn;
        for (int i = 0; i < numbers.size(); i++) {
            nn.push_back(Number(i+1, numbers[i]));
        }
        std::sort(nn.begin(), nn.end(), compare_number);
        int beg = 0;
        int end = nn.size() - 1;

        while (beg < end) {
            int sum = nn[beg].value + nn[end].value;
            if (sum == target) {
                int small = std::min(nn[beg].index, nn[end].index);
                int big = std::max(nn[beg].index, nn[end].index);
                vector<int> ret;
                ret.push_back(small);
                ret.push_back(big);
                return ret;
            } else if (sum < target) {
                beg++;
            } else {
                end--;
            }
        }
    }
};

public class Solution {
    public int[] twoSum(int[] numbers, int target) {
        HashMap<Integer, Integer> map = new HashMap<Integer, Integer>();
        int n = numbers.length;
        int[] result = new int[2];
        for (int i = 0; i < numbers.length; i++) {
            if (map.containsKey(target - numbers[i])) {
                result[0] = map.get(target - numbers[i]) + 1;
                result[1] = i + 1;
                break;
            } else {
                map.put(numbers[i], i);
            }
        }
        return result;
    }
}
