题目

  • 计算一个字符串的最长回文子字符串 
Sample Input : aaabaaaa
Sample Output: aaabaaa

思路

  • 这个题目可以看做是成名算法了吧.叫做Manacher's Algorithm,肯定不是一般人随便就 能想到的
  • Manacher's 的算法复杂度能够达到匪夷所思的O(n), 我们来看看怎么解的吧.很炫

解答

public class Solution {
    private String insertSharp(String s) {
        StringBuffer sb = new StringBuffer();
        sb.append("^#");
        for (int i = 0; i < s.length(); i++) {
            sb.append(s.charAt(i));
            sb.append('#');
        }
        sb.append('$');

        return sb.toString();
    }
    public String longestPalindrome(String s) {
        String newStr = insertSharp(s);

        int[] index = new int[newStr.length()];
        int maxPos = 0;
        int maxCenter = 0;

        for (int i = 1; i < newStr.length() - 1; i++) {
            int curV = 0;
            if (i < maxPos) {
                curV = Math.min(index[2 * maxCenter - i] , (maxPos - i));
            }

            while (newStr.charAt(i + curV + 1) == newStr.charAt(i - curV - 1)) {
                ++curV;
            }
            index[i] = curV;

            if (i + curV > maxPos) {
                maxPos = i + curV;
                maxCenter = i;
            }
        }

        int maxIndex = 0;
        for (int i = 1; i < index.length; i++) {
            if (index[i] > index[maxIndex]) {
                maxIndex = i;
            }
        }

        StringBuffer sb = new StringBuffer();
        for (int i = maxIndex - index[maxIndex]; i <= maxIndex + index[maxIndex]; i++) {
            if (newStr.charAt(i) != '#' && newStr.charAt(i) != '^' && newStr.charAt(i) != '$') {
                sb.append(newStr.charAt(i));
            }
        }

        System.out.println("Result is:" + sb.toString());
        return sb.toString();
    }
}
