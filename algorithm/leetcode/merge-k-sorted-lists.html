/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
    ListNode* mergeTwoLists(ListNode* lista, ListNode* listb) {
        if (lista == NULL) {
            return listb;
        }
        if (listb == NULL) {
            return lista;
        }
        ListNode* head = new ListNode(INT_MIN);
        ListNode* ret = head;

        while (lista || listb) {
            int va = (lista != NULL) ? lista->val : INT_MAX;
            int vb = (listb != NULL) ? listb->val : INT_MAX;
            if (va < vb) {
                head -> next = lista;
                lista = lista->next;
                head = head->next;
            } else {
                head->next = listb;
                listb = listb->next;
                head = head->next;
            }
        }
        delete ret;
        return ret->next;
    }
public:
    ListNode *mergeKLists(vector<ListNode *> &lists) {
        int len = lists.size();
        ListNode* ret  = NULL;
        for (int i = 0; i < len; i++) {
            ret = mergeTwoLists(ret, lists[i]);
        }
        return ret;
    }
};
