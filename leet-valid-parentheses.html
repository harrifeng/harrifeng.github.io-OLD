class Solution {
public:
    bool isValid(string s) {
        int len = s.length();
        if (len == 0) {
            return true;
        }
        vector<int> sta;
        sta.push_back(s[0]);
        for (int i = 1; i < len; i++) {
            if (s[i] == '(' || s[i] == '[' || s[i] == '{') {
                sta.push_back(s[i]);
            } else if (s[i] == ')' && sta.back() != '(') {
                return false;
            } else if (s[i] == ']' && sta.back() != '[') {
                return false;
            } else if (s[i] == '}' && sta.back() != '{') {
                return false;
            } else {
                sta.pop_back();
            }
        }
        if (sta.empty()) {
            return true;
        }
        return false;
    }
};
